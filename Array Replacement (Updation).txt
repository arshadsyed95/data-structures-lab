#include <stdio.h>

int main()
{
    int a[100], n, i, pos, item;

    printf("Enter size: ");
    scanf("%d", &n);

    for(i=0; i<n; i++)
        scanf("%d", &a[i]);

    printf("Enter position and new element: ");
    scanf("%d%d", &pos, &item);

    a[pos-1] = item;

    printf("Array after replacement: ");
    for(i=0; i<n; i++)
        printf("%d ", a[i]);

    return 0;
}
