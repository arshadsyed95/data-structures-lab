#include <stdio.h>

int main()
{
    int a[100], n, i, pos, item;

    printf("Enter size: ");
    scanf("%d", &n);

    for(i=0; i<n; i++)
        scanf("%d", &a[i]);

    printf("Enter position and element: ");
    scanf("%d%d", &pos, &item);

    for(i=n; i>=pos-1; i--)
        a[i+1] = a[i];

    a[pos-1] = item;
    n++;

    printf("Array after insertion: ");
    for(i=0; i<n; i++)
        printf("%d ", a[i]);

    return 0;
}
